63  _<ABasicListEditor> buildContainer() {

65  auto list = _new<AListView>(mModel);
66  list with_style { ass::MinSize { 200_dp, {} } };

68  auto modifyButton = mModifyCallback ? _new<AButton>("Modify...").connect(&AView::clicked, c, [callback = mModifyCallback, list] {
69  callback(list->getSelectionModel().first());
70  }) : nullptr;
71 
72  _<AButton> removeButton;
73  if (auto model = _cast<IRemovableListModel<AString>>(mModel)) {
74  removeButton = _new<AButton>("Remove").connect(&AView::clicked, c, [list, model] {
75  model->removeItems(list->getSelectionModel());
76  });
77  }
78 
79  if (mModel->listSize() != 0) {
80  list->selectItem(0);
81  } else {
82  AUI_NULLSAFE(removeButton)->disable();
83  AUI_NULLSAFE(modifyButton)->disable();
84  }
85  auto updateEnabledState = [list, modifyButton, removeButton]() {
86  auto& s = list->getSelectionModel().getIndices();
87  AUI_NULLSAFE(modifyButton)->setDisabled(s.empty());
88  AUI_NULLSAFE(removeButton)->setDisabled(s.empty());
89  };
90  connect(list->selectionChanged, c, updateEnabledState);
91  connect(mModel->dataRemoved, c, updateEnabledState);
92  connect(mModel->dataInserted, c, updateEnabledState);
93 
94  c->setContents(Horizontal {

96  Vertical {
