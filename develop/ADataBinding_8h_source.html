398 template<typename Klass1, typename View, typename Model, typename ModelField, typename GetterRV, typename SetterArg>
399 _<Klass1> operator&&(const _<Klass1>& modelBinding, const ADataBindingLinker<Model, View, ModelField, GetterRV, SetterArg>& linker) {
400  union converter {
401  unsigned i;
402  decltype(linker.getField()) p;
403  };
404  if (linker.getGetter()) {
405  AObject::connect(modelBinding.get()->*(linker.getGetter()), linker.getBinder(), [modelBinding, linker](const GetterRV& data) {
406  AUI_ASSERTX(&linker.getBinder()->getEditableModel(), "please setModel for ADataBinding");
407  modelBinding->setSignalsEnabled(false);
408  linker.getBinder()->getEditableModel().*(linker.getField()) = data;
409  converter c;
410  c.p = linker.getField();
411  linker.getBinder()->notifyUpdate(modelBinding.get(), c.i);
412  modelBinding->setSignalsEnabled(true);
413  });
414  }
415 
416  if constexpr (aui::convertible_to<decltype(linker.getSetterFunc()), bool>) {
417  if (!bool(linker.getSetterFunc())) {
418  return modelBinding;
419  }
420  }
421  linker.getBinder()->addObserver([modelBinding, linker](const Model& model, unsigned field) {
422  converter c;
423  c.p = linker.getField();
424  if (c.i == field || field == -1) {
425  if (modelBinding.get() != linker.getBinder()->getExclusion()) {
426  std::invoke(linker.getSetterFunc(), modelBinding.get(), model.*(linker.getField()));
427  }
428  }
429  });
430 
431  return modelBinding;
432 }

445 template<typename View, typename Model, typename Data, typename Projection>
446 _<View> operator&&(const _<View>& object, const ADataBindingLinker2<Model, Data, Projection>& linker) {
447  ADataBindingDefault<View, Data>::setup(object);
448 
449  constexpr bool is_default_projection = std::is_same_v<Projection, std::nullptr_t>;
450  using projection_deduced = std::conditional_t<is_default_projection, aui::identity, Projection>;
451  static_assert(std::is_invocable_v<projection_deduced, Data>, "projection is expected to accept Data value from model");
452  using data_deduced = std::decay_t<std::invoke_result_t<projection_deduced, Data>>;
453 
454  using getter = ASignal<Data>(View::*);
455  using setter = aui::member<decltype(ADataBindingDefault<View, data_deduced>::getSetter())>;
456 
457  using setter_ret = typename setter::return_t;
458  using setter_args = typename setter::args;
459 
460  using my_pointer_to_member = typename aui::detail::pointer_to_member<View, setter_ret>;
461 
462  using pointer_to_setter = decltype(my_pointer_to_member::with_args(std::declval<setter_args>()));
463 
464  if constexpr (is_default_projection) {
465  getter g = nullptr;
466  // getter is optional.
467  if constexpr (requires { ADataBindingDefault<View, Data>::getGetter(); }) {
468  if constexpr (requires { std::invoke(ADataBindingDefault<View, Data>::getGetter(), *object).changed; }) {
469  // AProperty.
470  g = (getter) &std::invoke(ADataBindingDefault<View, Data>::getGetter(), *object).changed;
471  } else {
472  g = (getter) &std::invoke(ADataBindingDefault<View, Data>::getGetter(), *object);
473  }
474  }
475  auto s = static_cast<pointer_to_setter>(ADataBindingDefault<View, Data>::getSetter());
476 
477  AUI_ASSERTX(g != nullptr || s != nullptr, "ADataBindingDefault is not defined for View, Data");
478 
479  object && (*linker.getBinder())(linker.getField(), g, s);
480  } else {
481  object && (*linker.getBinder())(linker.getField(), [projection = linker.getProjection()](View& v, const Data& d) {
482  auto s = static_cast<pointer_to_setter>(ADataBindingDefault<View, data_deduced>::getSetter());
483  std::invoke(s, &v, projection(d));
484  });
485  }
486  return object;
487 }
