84  static_assert(::aui::jni::convertible<ret_t>, "return type is required to be convertible"); \
85  auto clazz = getClass(); \

87  const char* signature = ::aui::jni::signature_v<ret_t (AUI_PP_FOR_EACH(AUI_JNI_INTERNAL_OMIT_BRACES, _, args))>; \
88  static auto methodId = e->GetStaticMethodID(clazz, #name, signature); \
89  if (methodId == 0) { \
90  throw AException("no such static jni method: {} {}"_format(#name, signature)); \
91  } \
92  return ::aui::jni::callStaticMethod<ret_t>(clazz, methodId AUI_PP_FOR_EACH(AUI_JNI_INTERNAL_OMIT_BRACES_CONTENTS, _, args)); \
