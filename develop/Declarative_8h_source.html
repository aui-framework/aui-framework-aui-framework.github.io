125  template<typename Layout, aui::derived_from<AViewContainer> Container = AViewContainer>
126  struct layouted_container_factory_impl {
127  private:
128  AVector<View> mViews;
129 
130  public:
131  template<typename... Views>
132  layouted_container_factory_impl(Views&&... views) {
133  mViews.reserve(sizeof...(views));
134  static_assert((ValidLayoutItem<Views> && ...),
135  "One of the items passed to declarative container is not valid. "
136  "Please check your compiler's diagnostics on constraint satisfaction.");
137  aui::parameter_pack::for_each([this]<ValidLayoutItem Item>(Item&& item) {
138  constexpr bool isView = LayoutItemView<Item>;
139  constexpr bool isViewGroup = LayoutItemViewGroup<Item>;
140  constexpr bool isInvokable = LayoutItemViewFactory<Item>;
141 
142  if constexpr (isViewGroup) {
143  auto asViewGroup = ViewGroup(item);
144  mViews << std::move(asViewGroup);
145  } else if constexpr (isView) {
146  auto asView = View(item);
147  mViews << std::move(asView);
148  } else if constexpr(isInvokable) {
149  mViews << item();
150  }
151  }, std::forward<Views>(views)...);
152  }
153 
154  _<Container> operator()() {
155  auto c = _new<Container>();
156  if constexpr(!std::is_same_v<Layout, std::nullopt_t>) {
157  c->setLayout(std::make_unique<Layout>());
158  }
159  c->setViews(std::move(mViews));
160  return c;
161  }
162  };
