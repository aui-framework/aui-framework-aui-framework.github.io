74 template<aui::invocable Callback>
75 void executeOnIoThreadBlocking(Callback&& callback) {
76 if (AThread::current() == mThread) {
77 callback();
78 return;
79 }
80
81 AFuture<> cs;
82 mMessageQueue.enqueue([&] {
83 callback();
84 cs.supplyValue();

86 mNotifyEvent.set();
