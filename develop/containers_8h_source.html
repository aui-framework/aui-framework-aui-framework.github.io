35 static constexpr std::size_t ceilPower2(std::size_t size) {
36 switch (size) {
37 case 0: return 0;
38 case 1: return 1;
39 default: break;
40 }
41
42 size -= 1;
43 std::size_t power = 1;
44 while (size != 0) {
45 size >>= 1;
46
47 if (size == 0) {
48 break;
49 }
50 ++power;
51 }
52 return std::size_t(1) << power;
53 }
54
55 static_assert(ceilPower2(228) == 256, "check ceilPower2");
56 static_assert(ceilPower2(256) == 256, "check ceilPower2");
57 static_assert(ceilPower2(512) == 512, "check ceilPower2");
58 static_assert(ceilPower2(0) == 0, "check ceilPower2");
59 static_assert(ceilPower2(1) == 1, "check ceilPower2");
60 static_assert(ceilPower2(2) == 2, "check ceilPower2");

62 template<typename T, typename OtherIterator>
63 auto insert_no_growth(T*& vectorEnd, T* at, OtherIterator begin, OtherIterator end) {
64 auto distance = std::distance(begin, end);
65
66 aui::range insertTargetRange(at, at + distance);
67
68 // shift elements to the right
69 aui::range shiftRange(at, vectorEnd);
70 if (!shiftRange.empty()) {
71 auto shiftFrom = std::prev(shiftRange.end());
72 auto shiftTo = shiftFrom + distance;
73 for (; shiftFrom >= shiftRange.begin(); --shiftFrom, --shiftTo) {
74 if (shiftTo >= vectorEnd) {
75 new (shiftTo) T(std::move(*shiftFrom));
76 } else {
77 *shiftTo = std::move(*shiftFrom);
78 }
79 }
80 }
81
82 auto it = at;
83 for (; it < vectorEnd && begin != end; ++it, ++begin) {
84 *it = *begin;
85 }
86
87 for (; begin != end; ++it, ++begin) {
88 new (it) T(*begin);
89 }
90
91 vectorEnd += distance;
92 return at;
93 }

95 template<typename T, typename OtherIterator>
96 auto erase(T* vectorBegin, T*& vectorEnd, OtherIterator begin, OtherIterator end) {
97 if (begin == end) {
98 return std::prev(vectorEnd);
99 }
100
101 auto eraseIt = begin;
102 auto remainingValueIt = end;
103
104 // move values to the left of erase range
105 for (; remainingValueIt != vectorEnd; ++eraseIt, ++remainingValueIt) {
106 *eraseIt = std::move(*remainingValueIt);
107 }
108 // destruct remaining values
109 std::size_t destructedValuesCounter = 0;
110 for (; eraseIt != vectorEnd; ++eraseIt, ++destructedValuesCounter) {
111 eraseIt->~T();
112 }
113 vectorEnd -= destructedValuesCounter;
114 return vectorBegin;
115 }
116 }

203 for (auto it = container.begin(); it != container.end(); ++it) {
204 if (*it == value) {
205 auto index = std::distance(container.begin(), it);

208 }
