199 _<AOverlappingSurface> createOverlappingSurface(const glm::ivec2& position,
200 const glm::ivec2& size,
201 bool closeOnClick = true) {
202 return createOverlappingSurface([&](unsigned attempt) -> AOptional<glm::ivec2> {
203 switch (attempt) {
204 case 0: return position;
205 case 1: return glm::clamp(position, {0, 0}, {getSize() - size});
206 default: return std::nullopt;
207 }
208 }, size, closeOnClick);
209 }
210
221 _<AOverlappingSurface> createOverlappingSurface(const std::function<AOptional<glm::ivec2>(unsigned)>& positionFactory,
222 const glm::ivec2& size,
223 bool closeOnClick = true) {
224 glm::ivec2 position = {0, 0};
225 auto maxPos = getSize() - size;
226 for (unsigned index = 0; ; ++index) {
227 auto optionalPosition = positionFactory(index);
228 if (optionalPosition) {
229 position = *optionalPosition;
230
231 if (position.x >= 0 && position.y >= 0 && glm::all(glm::lessThan(position, maxPos))) {
232 break;
233 }
234 } else {
235 break;
236 }
237 }
238
239 auto tmp = createOverlappingSurfaceImpl(position, size);
240 tmp->mParentWindow = this;
241 tmp->mCloseOnClick = closeOnClick;
242 mOverlappingSurfaces << tmp;
243 return tmp;
244 }
245 void closeOverlappingSurface(AOverlappingSurface* surface) {
246 if (mOverlappingSurfaces.erase(aui::ptr::fake(surface)) > 0) {
247 closeOverlappingSurfaceImpl(surface);
248 }
249 }
