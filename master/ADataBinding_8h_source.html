268 template<typename ModelField, typename FieldObserver>
269 void addObserver(ModelField(Model::*field), FieldObserver&& observer) {
270 mLinkObservers << [observer = std::forward<FieldObserver>(observer), field] (const Model& model, unsigned index) {
271 union converter {
272 unsigned i;
273 decltype(field) p;
274 } c;
275 c.p = field;
276 if (c.i == index || index == -1) {
277 observer(model.*field);
278 }
279 };
280 if (mModel) {
281 mLinkObservers.last()(*mModel, -1);
282 }
283 }

293template<typename Klass1, typename Klass2, typename Model, typename ModelField, typename GetterRV, typename SetterArg>
294_<Klass1> operator&&(const _<Klass1>& object, const ADataBindingLinker<Model, Klass2, ModelField, GetterRV, SetterArg>& linker) {
295 union converter {
296 unsigned i;
297 decltype(linker.getField()) p;
298 };
299 if (linker.getGetter()) {
300 AObject::connect(object.get()->*(linker.getGetter()), linker.getBinder(), [object, linker](const GetterRV& data) {
301 AUI_ASSERTX(&linker.getBinder()->getEditableModel(), "please setModel for ADataBinding");
302 object->setSignalsEnabled(false);
303 linker.getBinder()->getEditableModel().*(linker.getField()) = data;
304 converter c;
305 c.p = linker.getField();
306 linker.getBinder()->notifyUpdate(object.get(), c.i);
307 object->setSignalsEnabled(true);
308 });
309 }
310
311 if (linker.getSetterFunc()) {
312 linker.getBinder()->addObserver([object, linker](const Model& model, unsigned field) {
313 converter c;
314 c.p = linker.getField();
315 if (c.i == field || field == -1) {
316 if (object.get() != linker.getBinder()->getExclusion()) {
317 (object.get()->*(linker.getSetterFunc()))(model.*(linker.getField()));
318 }
319 }
320 });
321 }
322
323 return object;
324}
