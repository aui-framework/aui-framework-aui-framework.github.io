46 template<enum_t value>
47 static AString valueName() {
48#if AUI_COMPILER_MSVC
49 AString s = __FUNCSIG__;
50 AString::iterator end = s.begin() + s.rfind('>');
51 AString::iterator begin = (std::find_if(std::make_reverse_iterator(end), s.rend(), [](wchar_t c) {
52 return c == ':' || c == '<';
53 })).base();
54
55 AString result(begin, end);
56#else
57 AString s = __PRETTY_FUNCTION__;
58#if AUI_COMPILER_CLANG
59 auto end = s.rfind(']');
60#else
61 auto end = s.rfind(';');
62#endif
63 size_t begin = s.rfind("value =", end);
64 if (begin == AString::NPOS) {
65 begin = s.rfind('[', end) + 1;
66 } else {
67 begin += 8;
68 }
69 AString result = {s.begin() + begin, s.begin() + end};
70
71
72 for (size_t p; (p = result.find("::")) != AString::NPOS;) {
73 result = result.substr(p + 2);
74 }
75
76#endif
77 return result;
78 }

99 static enum_t byName(const AString& name) {
100 if (auto c = all().contains(name.uppercase())) {
101 return c->second;
102 }
103 return (enum_t)0;
104 }
