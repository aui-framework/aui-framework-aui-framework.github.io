22template<typename T, typename Layout>
23class AForEachUI: public AViewContainer, public AListModelObserver<T>::IListModelListener {
24public:
25 using List = _<IListModel<T>>;
26 using Factory = std::function<_<AView>(const T& value, size_t index)>;
27private:
28 _<AListModelObserver<T>> mObserver;
29
30 Factory mFactory;
31
32public:
33 AForEachUI(const List& list):
34 mObserver(_new<AListModelObserver<T>>(this)) {
35 setLayout(_new<Layout>());
36 setModel(list);
37 }
38
39 void setModel(const List& list) {
40 mObserver->setModel(list);
41 }
42
43 void insertItem(size_t at, const T& value) override {
44 if (!mFactory) {
45 return;
46 }
47 addView(at, mFactory(value, at));
48 }
49
50 void updateItem(size_t at, const T& value) override {
51 // TODO optimize
52 removeView(at);
53 addView(at, mFactory(value, at));
54 }
55
56 void removeItem(size_t at) override {
57 removeView(at);
58 }
59
60 void onDataCountChanged() override {
61 AUI_NULLSAFE(AWindow::current())->flagUpdateLayout();
62 }
63
64 void onDataChanged() override {
65 updateLayout();
66 }
67
68 void operator-(const Factory& f) {
69 mFactory = f;
70 setModel(mObserver->getModel());
71 }
72};
