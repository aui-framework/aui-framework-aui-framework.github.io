237// Because of MSVC treating a token with a comma in it as a single token when
238// passed to another macro, we need to force it to evaluate it as multiple
239// tokens. We do that by using a "IDENTITY(MACRO PARENTHESIZED_ARGS)" macro. We
240// define one per possible macro that relies on this behavior. Note "_Args" must
241// be parenthesized.
242#define AUI_PP_INTERNAL_INTERNAL_16TH(_1, _2, _3, _4, _5, _6, _7, _8, _9, \
243 _10, _11, _12, _13, _14, _15, _16, \
244 ...) \
245 _16
246#define AUI_PP_INTERNAL_16TH(_Args) \
247 AUI_PP_IDENTITY(AUI_PP_INTERNAL_INTERNAL_16TH _Args)
