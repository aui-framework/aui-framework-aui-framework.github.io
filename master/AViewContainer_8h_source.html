178 template<aui::predicate<_<AView>> Callback>
179 bool getViewAtRecursive(glm::ivec2 pos, const Callback& callback, ABitField<AViewLookupFlags> flags = AViewLookupFlags::NONE) {
180 _<AView> possibleOutput; // for case if anyone does not consumesClick
181 auto process = [&](const _<AView>& view) {
182 if (callback(view))
183 return true;
184 if (auto container = _cast<AViewContainer>(view)) {
185 if (container->getViewAtRecursive(pos - view->getPosition(), callback, flags)) {
186 return true;
187 }
188 }
189 return false;
190 };
191
192 for (const auto& view : aui::reverse_iterator_wrap(mViews)) {
193 auto targetPos = pos - view->getPosition();
194
195 if (targetPos.x < 0 || targetPos.y < 0 || targetPos.x >= view->getSize().x || targetPos.y >= view->getSize().y) {
196 continue;
197 }
198 if (!flags.test(AViewLookupFlags::IGNORE_VISIBILITY)) {
199 if (view->getVisibility() == Visibility::GONE ||
200 view->getVisibility() == Visibility::UNREACHABLE) {
201 continue;
202 }
203 }
204
205 if (view->consumesClick(targetPos)) {
206 if (flags.test(AViewLookupFlags::IGNORE_VISIBILITY) || (view->getVisibility() != Visibility::GONE &&
207 view->getVisibility() !=
208 Visibility::UNREACHABLE)) {
209 if (process(view)) {
210 return true;
211 }
212
213 if (flags.test(AViewLookupFlags::ONLY_ONE_PER_CONTAINER)) {
214 return false;
215 }
216 }
217 } else {
218 if (possibleOutput == nullptr) {
219 possibleOutput = view;
220 }
221 }
222 }
223 if (possibleOutput) {
224 return process(possibleOutput);
225 }
226
227 return false;
228 }

236 template<aui::predicate<_<AView>> Callback>
237 bool visitsViewRecursive(Callback&& callback, ABitField<AViewLookupFlags> flags = AViewLookupFlags::NONE) {
238 for (auto it = mViews.rbegin(); it != mViews.rend(); ++it) {
239 auto view = *it;
240 if (flags.test(AViewLookupFlags::IGNORE_VISIBILITY) || (view->getVisibility() != Visibility::GONE && view->getVisibility() != Visibility::UNREACHABLE)) {
241 if (callback(view))
242 return true;
243 if (auto container = _cast<AViewContainer>(view)) {
244 if (container->visitsViewRecursive(callback, flags)) {
245 return true;
246 }
247 }
248 if (flags.test(AViewLookupFlags::ONLY_ONE_PER_CONTAINER)) {
249 break;
250 }
251 }
252 }
253 return false;
254 }
