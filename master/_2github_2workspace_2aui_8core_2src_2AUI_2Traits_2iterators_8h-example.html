template <typename T>
struct reverse_iterator_wrap {
private:
const T& mIterable;
public:
reverse_iterator_wrap(const T& mIterable) : mIterable(mIterable) {}
auto begin() {
return mIterable.rbegin();
}
auto end() {
return mIterable.rend();
}
};
